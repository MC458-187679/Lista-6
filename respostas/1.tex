\begin{theorem}[subestrutura ótima]
    Seja $C = [c_1, \ldots, c_n]$ uma lista de $n$ cortes com número de trilhos mínimo e seja $C' = [c_1, \ldots, c_m]$, com $m \leq n$, a sublista contínua com soma mais próxima de $M$\red{, mas menor}. Então, a sublista $C - C' = [c_{m + 1}, \ldots, c_n]$ são cortes com número mínimo de trilhos.
\end{theorem}

\begin{theorem}[escolha gulosa]
    Seja $i \leq k$ o maior segmento com $t_i > 0$. Se ele existir, então existe uma lista de $n$ cortes $C = [c_1, \ldots, c_n]$ com menor número de trilhos tal que $c_1 = i$.
\end{theorem}
